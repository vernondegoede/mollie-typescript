# Lines

## Example Usage

```typescript
import { Lines } from "mollie/models/operations";

let value: Lines = {
  description: "nor gestate worriedly whoa aw plus",
  quantity: 289406,
  unitPrice: {
    currency: "EUR",
    value: "10.00",
  },
  discountAmount: {
    currency: "EUR",
    value: "10.00",
  },
  totalAmount: {
    currency: "EUR",
    value: "10.00",
  },
  vatAmount: {
    currency: "EUR",
    value: "10.00",
  },
};
```

# EnableMethodResponseBody

The payment method object.

## Example Usage

```typescript
import { EnableMethodResponseBody } from "mollie/models/operations";

let value: EnableMethodResponseBody = {
  minimumAmount: {
    currency: "EUR",
    value: "10.00",
  },
  maximumAmount: {
    currency: "EUR",
    value: "10.00",
  },
  pricing: [
    {
      fixed: {
        currency: "EUR",
        value: "10.00",
      },
    },
  ],
  issuers: [
    {
      id: "ideal_ABNANL2A",
      name: "ING Bank",
    },
  ],
};
```
